create database school; -- created the database called school 
use school; -- choosing the database 

-- creating the table for students and attendance
CREATE TABLE students (
    student_id INT PRIMARY KEY,
    name VARCHAR(50),
    class VARCHAR(10)
);

CREATE TABLE attendance (
    attendance_id INT PRIMARY KEY,
    student_id INT,
    date DATE,
    status VARCHAR(10),   -- Present / Absent
    FOREIGN KEY (student_id) REFERENCES students(student_id)
);

-- performing joins for two tables (Left Join)

select name,class,date,status
from students left join attendance on students.student_id = attendance.student_id;

-- using where caluse to filter the data retrieval
-- check the student attendance on 2025-01-10
select name,class,date,status
from students left join attendance on students.student_id = attendance.student_id where date like '%10';

-- check the student attendance on 2025-01-10 and students present on the day
select name,class,date,status
from students left join attendance on students.student_id = attendance.student_id where date like '%10' and status = 'Present';

-- check the student attendance on 2025-01-10 and students absent on the day
select name,class,date,status
from students left join attendance on students.student_id = attendance.student_id where date like '%10' and status = 'Absent';
